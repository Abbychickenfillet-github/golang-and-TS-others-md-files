## 問題分析

試用頁面與正式頁面共用 `useTimeLogStore`，且使用相同的 localStorage key `'timelog-storage'`，會造成資料混用。

### 當前問題
1. 相同的 localStorage key：`'timelog-storage'`
2. 相同的 Zustand store：`useTimeLogStore`
3. 資料會互相覆蓋：試用頁面會覆蓋正式頁面資料

### 解決方案

建議為試用頁面建立獨立的 store：

```javascript
// stores/useTrialTimeLogStore.js
'use client'

import { create } from 'zustand'
import { persist } from 'zustand/middleware'

// Trial TimeLog Zustand Store
export const useTrialTimeLogStore = create(
  persist(
    (set, get) => ({
      // 狀態
      title: '',
      desc: '',
      memo: '',
      startTime: null,
      endTime: null,
      steps: [],
      currentTime: null,
      isClient: false,
      lastStepTime: null,
      titleHistory: [],

      // Actions (與原本相同)
      setClient: (isClient) => set({ isClient }),
      updateCurrentTime: () => set({ currentTime: new Date() }),
      setTitle: (title) => set({ title }),
      setDesc: (desc) => set({ desc }),
      setMemo: (memo) => set({ memo }),
      setTitleHistory: (titleHistory) => set({ titleHistory }),
      
      startActivity: () => {
        const now = new Date()
        set({
          startTime: now,
          endTime: null,
          lastStepTime: now,
          steps: [],
        })
      },

      endActivity: () => {
        set({ endTime: new Date() })
      },

      addStep: () => {
        const state = get()
        const now = new Date()
        const newStep = {
          name: `步驟 ${state.steps.length + 1}`,
          description: '',
          startTime: now,
          endTime: null,
        }
        set({
          steps: [...state.steps, newStep],
          lastStepTime: now,
        })
      },

      endSubStep: (index) => {
        const state = get()
        const updatedSteps = state.steps.map((step, i) =>
          i === index ? { ...step, endTime: new Date() } : step
        )
        set({ steps: updatedSteps })
      },

      handleVoiceResult: (result) => {
        if (result.title) set({ title: result.title })
        if (result.desc) set({ desc: result.desc })
      },

      clearStorage: () => {
        set({
          title: '',
          desc: '',
          memo: '',
          startTime: null,
          endTime: null,
          steps: [],
          currentTime: null,
          isClient: false,
          lastStepTime: null,
          titleHistory: [],
        })
      },

      getElapsedMinutes: () => {
        const state = get()
        if (!state.startTime) return 0
        const endTime = state.endTime || state.currentTime
        if (!endTime) return 0
        return Math.floor((endTime.getTime() - state.startTime.getTime()) / 1000 / 60)
      },

      getActivityStatus: () => {
        const state = get()
        if (state.startTime && !state.endTime) return '進行中'
        if (state.endTime) return '已結束'
        return '準備中'
      },
    }),
    {
      // 使用不同的 localStorage key
      name: 'trial-timelog-storage', // 試用版專用 key
      partialize: (state) => ({
        title: state.title,
        desc: state.desc,
        startTime: state.startTime,
        endTime: state.endTime,
        steps: state.endTime ? state.steps : [],
        lastStepTime: state.lastStepTime,
      }),
      onRehydrateStorage: () => (state) => {
        if (state) {
          if (state.startTime && typeof state.startTime === 'string') {
            state.startTime = new Date(state.startTime)
          }
          if (state.endTime && typeof state.endTime === 'string') {
            state.endTime = new Date(state.endTime)
          }
          if (state.lastStepTime && typeof state.lastStepTime === 'string') {
            state.lastStepTime = new Date(state.lastStepTime)
          }
          if (state.steps && Array.isArray(state.steps)) {
            state.steps = state.steps.map((step) => ({
              ...step,
              startTime:
                step.startTime && typeof step.startTime === 'string'
                  ? new Date(step.startTime)
                  : step.startTime,
              endTime:
                step.endTime && typeof step.endTime === 'string'
                  ? new Date(step.endTime)
                  : step.endTime,
            }))
          }
        }
      },
    }
  )
)
```
