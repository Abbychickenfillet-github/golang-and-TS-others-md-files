```sql
-- 1. event_coupon_program 新欄位
ALTER TABLE event_coupon_program ADD COLUMN quantity_per_claim INT NOT NULL DEFAULT 1 COMMENT '每次購票/領取發放幾張' AFTER is_free;

ALTER TABLE event_coupon_claim ADD COLUMN quantity INT NOT NULL DEFAULT 1 COMMENT '持有數量（一筆一張，固定1）' AFTER holder_member_id;
ALTER TABLE event_coupon_claim ADD COLUMN source_order_item_id VARCHAR(36) DEFAULT NULL COMMENT '觸發來源訂單項目 ID' AFTER quantity;
ALTER TABLE event_coupon_claim DROP INDEX uk_coupon_claim_member_program;
CREATE INDEX idx_coupon_claim_holder ON event_coupon_claim (holder_member_id);
```

## 測試 3: 禮品券手動轉讓

### 3.1 API 轉讓（後端先測）
```bash
# 用帳號 A 轉 1 張給帳號 B
curl -X POST http://localhost:8080/api/v1/my-coupons/{claim_id}/transfer \
  -H "Authorization: Bearer {A的token}" \
  -H "Content-Type: application/json" \
  -d '{"recipient_email": "b@example.com", "quantity": 1}'
```
- [ ] 回傳 200 + 轉讓成功訊息

### 3.2 驗證 DB
```sql
-- 查被轉讓的那筆 claim
SELECT id, member_id, holder_member_id FROM event_coupon_claim WHERE id = '{claim_id}';
```
- [ ] `member_id` = A（不變，原始領取者）
- [ ] `holder_member_id` = B（新持有者）

### 3.3 驗證 transfer_log
```sql
SELECT entity_type, entity_id, from_member_id, to_member_id, quantity
FROM transfer_log
WHERE entity_type = 'coupon' AND entity_id = '{claim_id}';
```
- [ ] 有一筆紀錄
- [ ] `from_member_id` = A, `to_member_id` = B, `quantity` = 1
